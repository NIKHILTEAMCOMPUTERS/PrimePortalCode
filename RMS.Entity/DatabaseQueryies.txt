-- Trigger definition
CREATE TRIGGER accManagerHistory_InsertUpdate_trigger
AFTER INSERT OR UPDATE OF accountmanagerId ON public.project
FOR EACH ROW
EXECUTE FUNCTION accmanagerHistoryInsert();

-- Function definition
CREATE OR REPLACE FUNCTION public.accmanagerHistoryInsert()
RETURNS trigger
LANGUAGE plpgsql
AS $function$
BEGIN
    IF TG_OP = 'INSERT' OR (TG_OP = 'UPDATE' AND NEW.accountmanagerId IS DISTINCT FROM OLD.accountmanagerId) THEN
        INSERT INTO acprojecthistory (projectid, acmanagerid)
        VALUES (NEW.projectid, NEW.accountmanagerId);
    END IF;
    RETURN NEW;
END;
$function$;

CREATE OR REPLACE FUNCTION public.new_after_update_contractemployee()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
DECLARE
    related_contractend_date TIMESTAMP;
   related_contractstart_date timestamp;
    new_effectiveenddate TIMESTAMP; 
   new_effectivestartdate timestamp;
    related_contractemployee_Id integer;
    related_New_Id integer;
    Related_Old_Id integer;
   related_CreatedBy integer;
   is_released BOOLEAN := FALSE;
begin
	
      related_contractemployee_Id:=COALESCE(NEW.contractemployeeid, OLD.contractemployeeid);
      related_New_Id:=NEW.categorysubstatusid;
      Related_Old_Id:=OLD.categorysubstatusid;
     related_CreatedBy:=COALESCE(NEW.createdby, OLD.createdby);
  
  
    SELECT pc.contractenddate,pc.contractstartdate INTO related_contractend_date,related_contractstart_date
    FROM contractemployee ce
    JOIN projectcontract pc ON pc.contractid = ce.contractid
    WHERE ce.contractemployeeid = related_contractemployee_Id
    ORDER BY pc.contractenddate DESC
    LIMIT 1;
   
    IF TG_OP = 'INSERT' then    
    
    SELECT pc.contractenddate INTO related_contractend_date
    FROM contractemployee ce
    JOIN projectcontract pc ON pc.contractid = ce.contractid
    WHERE ce.contractemployeeid = NEW.contractemployeeid
    ORDER BY pc.contractenddate DESC
    LIMIT 1;
        new_effectiveenddate := related_contractend_date;
       new_effectivestartdate:=Now();
  INSERT INTO employeeprojecthistory (contractemployeeid, categorysubstatusid, effectivestartdate, effectiveenddate, actiontakenon, actiontakenby)
  VALUES (NEW.contractemployeeid, NEW.categorysubstatusid, new_effectivestartdate, new_effectiveenddate, NOW(),related_CreatedBy );
          
    END IF;
        
       
     IF TG_OP = 'UPDATE' THEN
        SELECT pc.contractenddate INTO related_contractend_date
        FROM contractemployee ce
        JOIN projectcontract pc ON pc.contractid = ce.contractid
        WHERE ce.contractemployeeid = OLD.contractemployeeid
        ORDER BY pc.contractenddate DESC
        LIMIT 1;
        SELECT (categorysubstatusname = 'Released') INTO is_released
        FROM categorysubstatus
        WHERE categorysubstatusid = NEW.categorysubstatusid;
      RAISE NOTICE 'Value: %', is_released;
     
        IF is_released THEN
            new_effectiveenddate := NOW();
           new_effectivestartdate :=related_contractstart_date;
        ELSE
            new_effectiveenddate := related_contractend_date;
           new_effectivestartdate :=related_contractstart_date;
        END IF;
       
        INSERT INTO employeeprojecthistory (contractemployeeid, categorysubstatusid, effectivestartdate, effectiveenddate, actiontakenon, actiontakenby)
        VALUES (OLD.contractemployeeid, NEW.categorysubstatusid, new_effectivestartdate, new_effectiveenddate, NOW(),related_CreatedBy );
       
       
     END IF;      

    
         

    RETURN NULL; 
END;
$function$

   CREATE OR REPLACE FUNCTION generate_month_year_series(start_date TIMESTAMP, end_date TIMESTAMP)
RETURNS TABLE(month_year TEXT) AS $$
BEGIN
    RETURN QUERY
    SELECT TO_CHAR(date_series, 'Mon-YY') AS month_year
    FROM generate_series(
             start_date,
             end_date,
             '1 month'::interval
         ) AS date_series;
END;
